-- Tạo cơ sở dữ liệu FastFoodManagement
CREATE DATABASE FastFoodManagement;
USE AA;
-- Tạo bảng CUSTOMER
CREATE TABLE CUSTOMER (
    MAKH INT AUTO_INCREMENT PRIMARY KEY,
    HoTen VARCHAR(100) NOT NULL,
    SDT VARCHAR(15) NOT NULL,
    DiemTichLuy INT DEFAULT 0
);
-- Tạo bảng EMPLOYEE
CREATE TABLE EMPLOYEE (
    MANV INT AUTO_INCREMENT PRIMARY KEY,
    HoTen VARCHAR(100) NOT NULL,
    ChucVu ENUM('Nhân viên bán hàng', 'Quản lý') NOT NULL,
    SDT VARCHAR(15) NOT NULL,
    NgayBatDau DATE NOT NULL
);
-- Tạo bảng PRODUCT
CREATE TABLE PRODUCT (
    MASP INT AUTO_INCREMENT PRIMARY KEY,
    TenSP VARCHAR(100) NOT NULL,
    DonGia DECIMAL(10, 2) NOT NULL,
    TinhTrang ENUM('Có sẵn', 'Hết hàng') DEFAULT 'Có sẵn',
    HinhAnh VARCHAR(255)
);
-- Tạo bảng INVOICE
CREATE TABLE INVOICE (
    MAHD INT AUTO_INCREMENT PRIMARY KEY,
    MAKH INT,
    MANV INT NOT NULL,
    ThoiGianTT DATETIME NOT NULL,
    TongTien DECIMAL(10, 2) NOT NULL,
    HinhThucTT ENUM('Tiền mặt', 'Thẻ', 'Chuyển khoản') NOT NULL,
    SoTienNhan DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (MAKH) REFERENCES CUSTOMER(MAKH),
    FOREIGN KEY (MANV) REFERENCES EMPLOYEE(MANV)
);
-- Tạo bảng INVOICE_DETAIL
CREATE TABLE INVOICE_DETAIL (
    MAHD INT NOT NULL,
    MASP INT NOT NULL,
    SoLuong INT NOT NULL,
    ThanhTien DECIMAL(10, 2) NOT NULL,
    PRIMARY KEY (MAHD, MASP),
    FOREIGN KEY (MAHD) REFERENCES INVOICE(MAHD),
    FOREIGN KEY (MASP) REFERENCES PRODUCT(MASP)
);
-- Tạo bảng PROMOTION
CREATE TABLE PROMOTION (
    MAKM INT AUTO_INCREMENT PRIMARY KEY,
    ThoiGianHL DATETIME NOT NULL,
    DieuKien TEXT NOT NULL,
    MucGiamGia DECIMAL(10, 2) NOT NULL
);
-- Tạo bảng CUSTOMER_PROMOTION
CREATE TABLE CUSTOMER_PROMOTION (
    MAKH INT NOT NULL,
    MAKM INT NOT NULL,
    PRIMARY KEY (MAKH, MAKM),
    FOREIGN KEY (MAKH) REFERENCES CUSTOMER(MAKH),
    FOREIGN KEY (MAKM) REFERENCES PROMOTION(MAKM)
);
